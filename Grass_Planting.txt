import java.util.*;
import java.io.*;
class Main{
  public static void main(String[] args) throws IOException{
    Scanner in = new Scanner(new File("planting.in"));
    int n = in.nextInt(),min=0;
    int[] arr = new int[n+1];
    for(int i =0;i<n-1;i++){
      int a = in.nextInt(), b =in.nextInt();
      arr[a]+=1;arr[b]+=1;
    }
    for (int i = 0; i <n; i++){ if (arr[i] > min){ min = arr[i]; }}
    System.out.println(min+1);
  }
}
